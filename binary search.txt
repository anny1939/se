#include <iostream>
#include <string>
using namespace std;

class Student {
  public:
  int rollNo;
  string name;
  float sgpa;
};

int searchName(Student arr[],int size,string target){
  int start = 0;
  int end = size-1;
  int mid = (start+end)/2;

  while(start <= end){
    if(arr[mid].name == target){
      return mid;
    }
    else if (arr[mid].name < target){
      start = mid + 1;
      mid = (start+end)/2;
      }
    else if (arr[mid].name > target){
      end = mid - 1;
      mid = (start + end)/2;
    }
    
  }
  return -1;
}

int main() {

  int const stuCount = 5;
  Student student[5] = {{101,"Alice",9.9},
  {105,"Bobby",4.5},
  {103,"Charles",6.9},
  {111,"Raj",8.9},
  {115,"Tej",8.8}};

  string show;
  cout<<"Enter the name you want to search: ";
  cin>>show;
  int index = searchName(student,stuCount,show);

  if (index != -1){
    cout<<"Roll No."<<student[index].rollNo<<"\t\t"<<"Name: "<<student[index].name<<"\t\t"<<"SGPA: "<<student[index].sgpa<<endl;
  }
  else{
    cout<<"No student found";
  }
  

  

  
}
